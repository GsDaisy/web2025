- 객체중첩
let student = {
name : "Doremi",
score : {
history : 85,
science : 94,
average : function() {
return (this.history + this.science) / 2
}
}
}
student.score.history
student.score.average()


- for...in 사용
let bag = {
type : "backpack",
color : "blue",
size : 15
}
for(key in bag) {
console.log('${key}'); // type, color, size
}
for(key in bag) {
console.log('${key} : ${bag[key]}');
}
