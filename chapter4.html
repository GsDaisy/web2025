function calcSum() { // 함수 선언
let sum = 0;
for(let i = 1; i <= 10; i++) {
sum += i;
}
return sum; // 호출했던 자리로 반환
}
console.log ('1부터 10까지 더하면 ${calcSum()}입니다.');
